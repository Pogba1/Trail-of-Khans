province_event = {
    id = 155008

    title = "Asturian Exodus Ends"
    desc = "The large scale emmigration of Asturians from a certain province has ceased."
	
	allow_multiple_instances = no	
	mean_time_to_happen = {
        days = 1
    }

    option = {
        name = "Allah is mighty!"
			add_province_modifier = {
				name = asturian_exodus_aftermath
				duration = -1
			}	
			remove_province_modifier = asturian_exodus_happened
    }
	
    trigger = {
	NOT = { has_province_modifier = asturian_exodus }
	NOT = { has_province_modifier = asturian_exodus_aftermath }
	has_province_modifier = asturian_exodus_happened
	pop_majority_culture = spanish
	is_core = ATR
	NOT = {is_core = CAT}
	owner = {
	tag = ADL
	}
    }   
}
