#Integrating the Ecuadorian identity
country_event = {
	id = 476001
	title = "EVTNAME476001" #What it means to be Columbian
	desc = "EVTDESC476001"
	picture = "Celebration"
	
	trigger = {
		tag = GCO
		any_owned_province = {
			is_core = ECU
		}
	}
	
	mean_time_to_happen = {
		months = 0.1
		}
	
	option = {
		name = "God grants victory to perseverance."
			limit = {
				any_owned_province = {
					is_core = ECU
				}	
			}
			any_owned = {
				remove_core = ECU
			}
			
			ai_chance = { factor = 1 }
	}
}
